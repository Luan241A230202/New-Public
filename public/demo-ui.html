Quick Start

JavaScript / Node.js
import { VideoShareAPI } from '@videoshare/api-client';

const api = new VideoShareAPI({
  apiKey: process.env.VIDEOSHARE_API_KEY
});

// Get trending videos
const trending = await api.videos.trending({ limit: 10 });

// Get ML recommendations
const recommendations = await api.ml.recommendations({
  userId: 'user-123',
  type: 'personalized'
});
